class Solution {
    public int maxSubArray(int[] nums) {
        //TC=O(N)//Optimal//Kadane's Algorithm
        int sum=0, maxsum=nums[0];
        for(int i=0;i<nums.length;i++){
                sum+=nums[i];
                if(sum>maxsum){
                    maxsum=sum;
                }
                if(sum<0){
                    sum=0;
                }
        }
        return maxsum;
    }
}
